<script>
$("#regioes").on("change", function(){
   
   var idEstado = $("#regioes").val();
   $.ajax({
       url:'entregador-list.php',
       type:'POST',
       data:{id:idEstado},
      
       success:function(data){
           $("#rota").css({'display':'block'});
           $("#rota").html(data);
       }
   })

});

</script>
